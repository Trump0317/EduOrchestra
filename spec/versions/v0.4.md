# v0.4 — REST API 暴露状态图 + React 前端全链路交互

> 目标：后端提供 REST API 操作状态图，前端单页应用可完成创建任务→答题→反馈的完整流程

## API 设计

```
POST /api/task               body: {goal}     → 201 + TaskState
GET  /api/task/{task_id}                       → 200 + TaskState
POST /api/task/{task_id}/answer body: {answers} → 200 + TaskState
```

TaskState 格式（`questions` 不暴露正确答案）：

```json
{
  "task_id": "uuid",
  "task_goal": "掌握二次函数",
  "status": "waiting_for_answer" | "completed",
  "plan": [{"title": "...", "desc": "..."}],
  "current_step": 0,
  "resources": [{"type": "video", "title": "...", "url": "..."}],
  "questions": [{"id": "q1", "content": "...", "options": ["A.", "B."], "kp": "..."}],
  "analytics": null | {"total_questions": 2, "correct_count": 1, ...},
  "feedback": null | {"summary": "...", "suggestion": "..."},
  "next_action": ""
}
```

前端交互流程：

```
输入目标 → [加载中] → 计划 + 资料 → 答题 → [加载中] → 分析 + 反馈
                                                          ↓ repeat    ↓ next
                                                      重新答题    新步骤资料 → 答题
```

### 实现

**routers/task.py 核心逻辑**：

```python
router = APIRouter(prefix="/api/task", tags=["task"])
_graph = build_graph()  # 模块级共享实例（MemorySaver 持状态）


@router.post("", status_code=201)
def create_task(body: TaskCreate) -> TaskResponse:
    """创建任务，首次 invoke 跑到 practice 中断，返回状态。"""
    task_id = str(uuid.uuid4())
    cfg = {"configurable": {"thread_id": task_id}}
    state = _graph.invoke(
        {
            "task_id": task_id,
            "task_goal": body.goal,
            "plan": [],
            "current_step": 0,
            "resources": [],
            "questions": [],
            "waiting_for_answer": False,
            "answers": [],
            "analytics": None,
            "feedback": None,
            "next_action": "",
        },
        cfg,
    )
    return _format_response(task_id, state)


def _get_state_or_404(task_id: str):
    """查状态，不存在时返回 404"""
    cfg = {"configurable": {"thread_id": task_id}}
    snapshot = _graph.get_state(cfg)
    if snapshot is None or not snapshot.values:
        raise HTTPException(404, detail="任务不存在")
    return snapshot

@router.get("/{task_id}")
def get_task(task_id: str):
    """查询任务当前状态。"""
    snapshot = _get_state_or_404(task_id)
    return _format_response(task_id, snapshot.values)


@router.post("/{task_id}/answer")
def submit_answer(task_id: str, body: AnswerBody):
    """提交答案，恢复图执行，返回新状态（可能又停在 practice）。"""
    from orchestrator.agents.practice import check_answer

    # 先查当前状态获取 questions（含正确答案）
    snapshot = _get_state_or_404(task_id)
    current_state = snapshot.values
    questions = {q["id"]: q for q in current_state.get("questions", [])}

    # 构造 answers（后端判题）
    answers = []
    for a in body.answers:
        q = questions.get(a.question_id)
        if not q:
            raise HTTPException(422, detail=f"题目 {a.question_id} 不存在")
        answers.append({
            "question_id": a.question_id,
            "student_answer": a.student_answer,
            "is_correct": check_answer(q, a.student_answer),
            "correct_answer": q["answer"],
        })

    _graph.update_state(cfg, {"answers": answers, "waiting_for_answer": False})
    state = _graph.invoke(None, cfg)
    return _format_response(task_id, state)
```

**响应格式化**（去掉 questions 中的 answer 字段）：

```python
def _format_response(task_id: str, state: dict) -> dict:
    return {
        "task_id": task_id,
        "task_goal": state.get("task_goal", ""),
        "status": "completed" if not state.get("waiting_for_answer") else "waiting_for_answer",
        "plan": state.get("plan", []),
        "current_step": state.get("current_step", 0),
        "resources": state.get("resources", []),
        "questions": [
            {"id": q["id"], "content": q["content"],
             "options": q["options"], "kp": q.get("kp", "")}
            for q in state.get("questions", [])
        ],
        "analytics": state.get("analytics"),
        "feedback": state.get("feedback"),
        "next_action": state.get("next_action", ""),
    }
```

### 实现

**TypeScript 类型定义**（内联在 App.tsx 中）：

```typescript
interface TaskState {
  task_id: string;
  task_goal: string;
  status: "waiting_for_answer" | "completed";
  plan: { title: string; desc: string }[];
  current_step: number;
  resources: { type: string; title: string; url: string }[];
  questions: { id: string; content: string; options: string[]; kp: string }[];
  analytics: Record<string, any> | null;
  feedback: { summary: string; suggestion: string } | null;
  next_action: string;
}
```

**App 状态设计**：

```typescript
// 用一个 state 管理全部
const [taskState, setTaskState] = useState<TaskState | null>(null);
const [loading, setLoading] = useState(false);
const [selectedAnswers, setSelectedAnswers] = useState<Record<string, string>>({});
const [phase, setPhase] = useState<"idle" | "ready" | "result">("idle");
```

**API 调用流程**：

```typescript
// 1. 创建任务
async function handleStart(goal: string) {
  setLoading(true);
  const resp = await fetch("/api/task", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ goal }),
  });
  const state = await resp.json();
  setTaskState(state);
  setPhase("ready");
  setLoading(false);
}

// 2. 提交答案
async function handleSubmit() {
  const answers = Object.entries(selectedAnswers).map(
    ([question_id, student_answer]) => ({ question_id, student_answer })
  );
  setLoading(true);
  const resp = await fetch(`/api/task/${taskState.task_id}/answer`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ answers }),
  });
  const state = await resp.json();
  setTaskState(state);
  setSelectedAnswers({});
  setPhase(state.status === "completed" ? "result" : "ready");
  setLoading(false);
}
```

**循环逻辑**：
- 提交答案后，后端返回新 state
- 如果 `status === "waiting_for_answer"`（repeat 或 next_step 到新步骤的题目），`setPhase("ready")` → 继续答题
- 如果 `status === "completed"`（全部完成），`setPhase("result")` → 显示完成页

**视图切换**（条件渲染，无路由）：

- **idle**（phase=idle, !taskState）：输入框 + 开始按钮
- **loading**（loading=true）：CSS spinner + 提示文字（"AI 正在生成学习计划..."或"正在分析答题结果..."）
- **ready**（phase=ready, taskState.waiting_for_answer）：
  - 面包屑：当前步骤/总步骤
  - 资料卡片（type 图标 + title + URL）
  - 题目列表（每个题目：题目文字 + 4 个 radio 选项）
  - "提交答案"按钮（全部选完才可点）
- **result**（phase=result, taskState.completed）：
  - 最终信息：所有步骤完成

**Style**：不改组件库，纯 CSS。单列布局 max-width 640px 居中，卡片式分组（每个区域一个白底卡片），移动端友好。
