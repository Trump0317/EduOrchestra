**llm.py 核心接口**：

```python
# server/orchestrator/llm.py
from langchain_openai import ChatOpenAI
from config import config

def get_llm() -> ChatOpenAI:
    """创建 LLM 客户端实例。缺 API Key 时抛出 ValueError。"""
    missing = config.validate()
    if missing:
        raise ValueError(f"缺少配置: {', '.join(missing)}。请检查 .env 文件。")
    kwargs = dict(model=config.LLM_MODEL, api_key=config.LLM_API_KEY)
    if config.LLM_BASE_URL:
        kwargs["base_url"] = config.LLM_BASE_URL
    return ChatOpenAI(**kwargs)

def llm_invoke(prompt: str) -> str:
    """调 LLM，返回纯文本。"""
    llm = get_llm()
    resp = llm.invoke(prompt)
    return resp.content

def llm_invoke_json(prompt: str) -> dict:
    """调 LLM，解析 JSON 输出。解析失败抛出 ValueError。"""
    import json, re
    text = llm_invoke(prompt)
    # 尝试提取 JSON 块（```json ... ``` 或直接的 {...}）
    match = re.search(r'\{.*\}', text, re.DOTALL)
    if match:
        return json.loads(match.group())
    raise ValueError(f"LLM 输出中未找到 JSON: {text[:200]}")
```

**prompt.py 核心接口**：

```python
# server/orchestrator/prompt.py
from pathlib import Path

PROMPT_DIR = Path(__file__).resolve().parent.parent.parent / "data" / "prompts"

def load_prompt(name: str) -> str:
    """加载 prompt 模板文件。name 不含扩展名，如 'planner'。"""
    path = PROMPT_DIR / f"{name}.md"
    if not path.exists():
        raise FileNotFoundError(f"Prompt 模板不存在: {path}")
    return path.read_text(encoding="utf-8")

def render_prompt(name: str, **kwargs) -> str:
    """加载并渲染 prompt 模板。"""
    template = load_prompt(name)
    return template.format(**kwargs)
```

**planner.py** 核心逻辑：

```python
# orchestrator/agents/planner.py
from orchestrator.state import AgentState
from orchestrator.llm import llm_invoke_json
from orchestrator.prompt import render_prompt

def planner_node(state: AgentState) -> dict:
    """调用 LLM 拆解学习目标为步骤序列。不修改 current_step。"""
    goal = state["task_goal"]
    prompt = render_prompt("planner", task_goal=goal)
    result = llm_invoke_json(prompt)
    # LLM 输出: {"steps": [{"title": "...", "desc": "..."}, ...]}
    return {"plan": result["steps"]}
```

**planner.md** prompt 设计思路：

- 任务：根据学习目标拆解为 2-5 个学习步骤
- 约束：步骤之间有逻辑递进，每步聚焦一个可验证的子目标
- 输出格式：严格 JSON `{"steps": [{"title": "...", "desc": "..."}]}`
- 领域限定：仅限高一数学内容

### 实现

**analytics.py** 核心逻辑：

```python
def analytics_node(state: AgentState) -> dict:
    answers = state["answers"]
    total = len(answers)
    correct = sum(1 for a in answers if a["is_correct"])
    accuracy = correct / total if total > 0 else 0

    # 统计层面（不需要 LLM）
    stats = {
        "total_questions": total,
        "correct_count": correct,
        "accuracy": accuracy,
    }

    # LLM 分析薄弱点（仅当有答错的题）
    if correct < total:
        wrong_details = [
            {"question": a["question_id"], "student_answer": a["student_answer"],
             "correct_answer": a["correct_answer"]}
            for a in answers if not a["is_correct"]
        ]
        prompt = render_prompt("analytics",
            task_goal=state["task_goal"],
            wrong_details=json.dumps(wrong_details, ensure_ascii=False),
        )
        llm_result = llm_invoke_json(prompt)
        # LLM 输出: {"weak_points": ["...", ...], "summary": "..."}
        stats["weak_points"] = llm_result.get("weak_points", [])
        stats["summary"] = llm_result.get("summary", "")
    else:
        stats["weak_points"] = []
        stats["summary"] = "全部正确，掌握良好"

    return {"analytics": stats}
```

**analytics.md** prompt 设计思路：
- 输入：学习目标 + 答错题目的详细信息（题目ID、学生答案、正确答案）
- 输出：薄弱知识点列表 + 简短总结
- 约束：知识点粒度适中，不泛泛也不过度细分

### 实现

**feedback.py** 核心逻辑：

```python
def feedback_node(state: AgentState) -> dict:
    analytics = state["analytics"]
    prompt = render_prompt("feedback",
        task_goal=state["task_goal"],
        accuracy=analytics["accuracy"],
        weak_points=json.dumps(analytics.get("weak_points", []), ensure_ascii=False),
        total_questions=analytics["total_questions"],
        correct_count=analytics["correct_count"],
    )
    result = llm_invoke_json(prompt)
    # LLM 输出: {"summary": "...", "suggestion": "...", "action": "repeat_step"|"next_step"}

    update = {
        "feedback": {"summary": result["summary"], "suggestion": result["suggestion"]},
        "next_action": result["action"],
    }

    # action 为 next_step 时递增 current_step
    if result["action"] == "next_step":
        update["current_step"] = state["current_step"] + 1

    return update
```

**feedback.md** prompt 设计思路：
- 输入：学习目标、正确率、薄弱点、答题统计
- 输出：反馈总结 + 建议 + 路由决策
- 路由规则（内置到 prompt）：正确率 < 60% → repeat_step，≥ 60% → next_step
- 反馈风格：鼓励性语言，具体指出改进方向
